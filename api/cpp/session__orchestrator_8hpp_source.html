249 auto handle(const core::events::trust &cmd) -> void
250 {
251 bool contact_exists = false;
252 try {
253 std::ignore = bridge_->lookup_contact(cmd.peer);
254 contact_exists = true;
255 } catch (const std::exception &e) {
256 spdlog::debug("Contact {} not found, will check discovered bundles: {}", cmd.peer, e.what());
257 }
258
259 if (contact_exists) {
260 if (not cmd.alias.empty()) {
261 handler_.handle(cmd);
262 spdlog::info("Updated alias for existing contact: {}", cmd.peer);
263 }
264 return;
265 }
266
267 auto bundle_iter = std::ranges::find_if(
268 discovered_bundles_, [&cmd](const discovered_bundle &bundle) { return bundle.rdx_fingerprint == cmd.peer; });
269
270 if (bundle_iter != discovered_bundles_.end()) {
271 auto session_result =
272 handler_.handle(core::events::establish_session{ .bundle_data = bundle_iter->bundle_base64 });
273 if (session_result and not cmd.alias.empty()) { handler_.handle(cmd); }
274 if (session_result) { emit_presentation_event(*session_result); }
275 } else {
276 spdlog::error(
277 "Cannot establish session with {}: identity not found in discovered bundles and no existing contact", cmd.peer);
278 }
279 }

397 auto handle(const core::events::transport::bytes_received &evt) noexcept -> void
398 {
399 try {
400 std::string json_str;
401 json_str.resize(evt.bytes.size());
402 std::ranges::transform(evt.bytes, json_str.begin(), [](std::byte byte) { return std::bit_cast<char>(byte); });
403
404 try {
405 auto parsed = nlohmann::json::parse(json_str);
406 if (not parsed.is_array() or parsed.empty() or not parsed[0].is_string()) {
407 nostr::events::incoming::unknown_protocol evt_inner{ json_str };
408 handler_.handle(evt_inner);
409 return;
410 }
411
412 const auto msg_type = parsed[0].get<std::string>();
413
414 if (msg_type == "OK") {
415 auto ok_msg = nostr::protocol::ok::deserialize(json_str);
416 if (ok_msg) {
417 tracker_->resolve(ok_msg->event_id, *ok_msg);
418 nostr::events::incoming::ok evt_inner{ *ok_msg };
419 handler_.handle(evt_inner);
420 } else {
421 nostr::events::incoming::unknown_protocol evt_inner{ json_str };
422 handler_.handle(evt_inner);
423 }
424 } else if (msg_type == "EOSE") {
425 auto eose_msg = nostr::protocol::eose::deserialize(json_str);
426 if (eose_msg) {
427 tracker_->resolve(eose_msg->subscription_id, *eose_msg);
428 nostr::events::incoming::eose evt_inner{ *eose_msg };
429 handler_.handle(evt_inner);
430 } else {
431 nostr::events::incoming::unknown_protocol evt_inner{ json_str };
432 handler_.handle(evt_inner);
433 }
434 } else if (msg_type == "EVENT" and parsed.size() >= 3) {
435 auto event_data = parsed[2];
436 auto kind_value = event_data["kind"].get<std::uint32_t>();
437
438 switch (static_cast<nostr::protocol::kind>(kind_value)) {
439 case nostr::protocol::kind::encrypted_message: {
440 nostr::events::incoming::encrypted_message evt_inner{ nostr::protocol::event_data{ .id = event_data["id"],
441 .pubkey = event_data["pubkey"],
442 .created_at = event_data["created_at"],
443 .kind = event_data["kind"],
444 .tags = event_data["tags"],
445 .content = event_data["content"],
446 .sig = event_data["sig"] } };
447
448 if (auto result = handler_.handle(evt_inner)) {
449 emit_presentation_event(*result);
450 if (result->should_republish_bundle) { handle(core::events::publish_identity{}); }
451 }
452 break;
453 }
454 case nostr::protocol::kind::bundle_announcement: {
455 nostr::events::incoming::bundle_announcement evt_inner{ nostr::protocol::event_data{ .id = event_data["id"],
456 .pubkey = event_data["pubkey"],
457 .created_at = event_data["created_at"],
458 .kind = event_data["kind"],
459 .tags = event_data["tags"],
460 .content = event_data["content"],
461 .sig = event_data["sig"] } };
462
463 if (auto result = handler_.handle(evt_inner)) {
464 std::visit(
465 [this](auto &&inner_evt) {
466 handle(inner_evt);
467 emit_presentation_event(inner_evt);
468 },
469 *result);
470 }
471 break;
472 }
473 case nostr::protocol::kind::identity_announcement: {
474 nostr::events::incoming::identity_announcement evt_inner{ nostr::protocol::event_data{
475 .id = event_data["id"],
476 .pubkey = event_data["pubkey"],
477 .created_at = event_data["created_at"],
478 .kind = event_data["kind"],
479 .tags = event_data["tags"],
480 .content = event_data["content"],
481 .sig = event_data["sig"] } };
482
483 handler_.handle(evt_inner);
484 break;
485 }
486 case nostr::protocol::kind::session_request: {
487 nostr::events::incoming::session_request evt_inner{ nostr::protocol::event_data{ .id = event_data["id"],
488 .pubkey = event_data["pubkey"],
489 .created_at = event_data["created_at"],
490 .kind = event_data["kind"],
491 .tags = event_data["tags"],
492 .content = event_data["content"],
493 .sig = event_data["sig"] } };
494
495 handler_.handle(evt_inner);
496 break;
497 }
498 case nostr::protocol::kind::node_status: {
499 nostr::events::incoming::node_status evt_inner{ nostr::protocol::event_data{ .id = event_data["id"],
500 .pubkey = event_data["pubkey"],
501 .created_at = event_data["created_at"],
502 .kind = event_data["kind"],
503 .tags = event_data["tags"],
504 .content = event_data["content"],
505 .sig = event_data["sig"] } };
506
507 handler_.handle(evt_inner);
508 break;
509 }
510 default: {
511 nostr::events::incoming::unknown_message evt_inner{ nostr::protocol::event_data{ .id = event_data["id"],
512 .pubkey = event_data["pubkey"],
513 .created_at = event_data["created_at"],
514 .kind = event_data["kind"],
515 .tags = event_data["tags"],
516 .content = event_data["content"],
517 .sig = event_data["sig"] } };
518
519 handler_.handle(evt_inner);
520 break;
521 }
522 }
523 } else {
524 nostr::events::incoming::unknown_protocol evt_inner{ json_str };
525 handler_.handle(evt_inner);
526 }
527 } catch (const std::exception &e) {
528 std::string error_msg(e.what());
529 if (error_msg.find("old counter") != std::string::npos
530 or error_msg.find("message with old") != std::string::npos) {
531 try {
532 auto parsed = nlohmann::json::parse(json_str);
533 if (parsed.is_array() and parsed.size() >= 3 and parsed[2].contains("pubkey")
534 and parsed[2].contains("id")) {
535 spdlog::debug("[session_orchestrator] Ignored duplicate message from {} (event: {})",
536 parsed[2]["pubkey"].get<std::string>().substr(0, 16),
537 parsed[2]["id"].get<std::string>().substr(0, 16));
538 } else {
539 spdlog::debug("[session_orchestrator] Ignored duplicate message: {}", error_msg);
540 }
541 } catch (...) {
542 spdlog::debug("[session_orchestrator] Ignored duplicate message: {}", error_msg);
543 }
544 } else {
545 spdlog::warn("[session_orchestrator] Failed to parse message: {} - Raw: {}", error_msg, json_str);
546 nostr::events::incoming::unknown_protocol evt_inner{ json_str };
547 handler_.handle(evt_inner);
548 }
549 }
550 } catch (const std::bad_alloc &e) {
551 spdlog::error("[session_orchestrator] Failed to process bytes_received event: {}", e.what());
552 }
553 }
