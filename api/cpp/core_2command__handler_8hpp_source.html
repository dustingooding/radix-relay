238 auto handle_impl(const events::chat &command) const -> void
239 {
240 std::ignore = initialized_;
241 if (command.contact.empty()) {
242 emit("Usage: /chat <contact>\n");
243 return;
244 }
245
246 try {
247 const auto contact = bridge_->lookup_contact(command.contact);
248 const auto display_name = contact.user_alias.empty() ? contact.rdx_fingerprint : contact.user_alias;
249
250 display_out_queue_->push(
251 events::enter_chat_mode{ .rdx_fingerprint = contact.rdx_fingerprint, .display_name = display_name });
252
253 constexpr std::uint32_t history_limit = 5;
254 const auto messages = bridge_->get_conversation_messages(contact.rdx_fingerprint, history_limit, 0);
255
256 if (not messages.empty()) {
257 display_out_queue_->push(events::display_message{
258 .message = fmt::format("--- Conversation History ({} messages) ---", messages.size()),
259 .contact_rdx = contact.rdx_fingerprint,
260 .timestamp = platform::current_timestamp_ms(),
261 .source_type = events::display_message::source::system });
262
263 for (auto it = messages.rbegin(); it != messages.rend(); ++it) {
264 const auto &msg = *it;
265
266 const auto direction_indicator = (msg.direction == signal::MessageDirection::Incoming) ? "← " : "→ ";
267 const auto sender_name = (msg.direction == signal::MessageDirection::Incoming) ? display_name : "You";
268
269 const auto formatted_message = fmt::format("{}{}: {}", direction_indicator, sender_name, msg.content);
270
271 const auto source_type = (msg.direction == signal::MessageDirection::Incoming)
272 ? events::display_message::source::incoming_message
273 : events::display_message::source::outgoing_message;
274
275 display_out_queue_->push(events::display_message{ .message = formatted_message,
276 .contact_rdx = contact.rdx_fingerprint,
277 .timestamp = msg.timestamp,
278 .source_type = source_type });
279 }
280
281 display_out_queue_->push(events::display_message{ .message = "--- End of History ---",
282 .contact_rdx = contact.rdx_fingerprint,
283 .timestamp = platform::current_timestamp_ms(),
284 .source_type = events::display_message::source::system });
285
286 const auto newest_timestamp = messages.front().timestamp;
287 bridge_->mark_conversation_read_up_to(contact.rdx_fingerprint, newest_timestamp);
288 } else {
289 bridge_->mark_conversation_read(contact.rdx_fingerprint);
290 }
291
292 emit("Entering chat with {}\n", display_name);
293 } catch (const std::exception & /*e*/) {
294 emit("Contact not found: {}\n", command.contact);
295 }
296 }
